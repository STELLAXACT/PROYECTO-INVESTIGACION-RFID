Sistema RFID con ESP32

Implementación de un sistema RFID utilizando ESP32 con conectividad WiFi para aplicaciones IoT. Incluye configuración de tarjetas y comunicación inalámbrica.

Descripción General del Proyecto

Este proyecto implementa un sistema RFID completo utilizando el microcontrolador ESP32, aprovechando sus capacidades de conectividad WiFi para crear una solución IoT robusta.

El sistema permite la lectura de tarjetas RFID, configuración remota via WiFi y procesamiento en tiempo real de los datos de identificación.

Ventajas del ESP32:

  • Conectividad WiFi y Bluetooth integrada
  • Procesador dual-core a 240MHz
  • Bajo consumo energético
  • Compatible con Arduino IDE
  • Múltiples interfaces de comunicación (SPI, I2C, UART)

Archivo: card_holder_configure.ino 

/*
 * Configuración de Tarjetas RFID para ESP32
 * Este código permite configurar y registrar nuevas tarjetas RFID
 * en el sistema mediante interfaz serie
 */

#include 
#include 
#include 
#include 

// Definición de pines para ESP32
#define RST_PIN    22
#define SS_PIN     5
#define SCK_PIN    18
#define MOSI_PIN   23
#define MISO_PIN   19

// Configuración WiFi
const char* ssid = "TU_RED_WIFI";
const char* password = "TU_PASSWORD";

// Inicialización del módulo RFID
MFRC522 mfrc522(SS_PIN, RST_PIN);

// Estructura para almacenar datos de tarjeta
struct CardData {
    byte uid[4];
    char name[32];
    bool active;
};

// Array para almacenar hasta 50 tarjetas
CardData registeredCards[50];
int cardCount = 0;

void setup() {
    Serial.begin(115200);
    
    // Inicializar SPI y RFID
    SPI.begin(SCK_PIN, MISO_PIN, MOSI_PIN, SS_PIN);
    mfrc522.PCD_Init();
    
    // Inicializar EEPROM
    EEPROM.begin(4096);
    
    // Cargar tarjetas desde EEPROM
    loadCardsFromEEPROM();
    
    // Conectar a WiFi
    connectToWiFi();
    
    Serial.println("=================================");
    Serial.println("Sistema de Configuración RFID ESP32");
    Serial.println("=================================");
    Serial.println("Comandos disponibles:");
    Serial.println("1 - Registrar nueva tarjeta");
    Serial.println("2 - Listar tarjetas registradas");
    Serial.println("3 - Eliminar tarjeta");
    Serial.println("4 - Modo lectura continua");
    Serial.println("=================================");
    
    delay(1000);
}

void loop() {
    // Verificar comandos por Serial
    if (Serial.available() > 0) {
        String command = Serial.readString();
        command.trim();
        
        switch (command.toInt()) {
            case 1:
                registerNewCard();
                break;
            case 2:
                listRegisteredCards();
                break;
            case 3:
                deleteCard();
                break;
            case 4:
                continuousReadMode();
                break;
            default:
                Serial.println("Comando no válido");
                break;
        }
    }
    
    delay(100);
}

void connectToWiFi() {
    Serial.print("Conectando a WiFi");
    WiFi.begin(ssid, password);
    
    while (WiFi.status() != WL_CONNECTED) {
        delay(500);
        Serial.print(".");
    }
    
    Serial.println();
    Serial.println("WiFi conectado!");
    Serial.print("IP address: ");
    Serial.println(WiFi.localIP());
}

void registerNewCard() {
    Serial.println("Acerca la tarjeta RFID para registrar...");
    
    // Esperar hasta que se detecte una tarjeta
    while (!mfrc522.PICC_IsNewCardPresent() || !mfrc522.PICC_ReadCardSerial()) {
        delay(50);
    }
    
    // Leer UID de la tarjeta
    byte cardUID[4];
    for (byte i = 0; i < 4; i++) {
        cardUID[i] = mfrc522.uid.uidByte[i];
    }
    
    // Verificar si la tarjeta ya está registrada
    if (isCardRegistered(cardUID)) {
        Serial.println("Esta tarjeta ya está registrada!");
        return;
    }
    
    // Solicitar nombre para la tarjeta
    Serial.print("Ingresa el nombre para esta tarjeta: ");
    while (!Serial.available()) {
        delay(100);
    }
    
    String cardName = Serial.readString();
    cardName.trim();
    
    // Registrar la nueva tarjeta
    if (cardCount < 50) {
        memcpy(registeredCards[cardCount].uid, cardUID, 4);
        cardName.toCharArray(registeredCards[cardCount].name, 32);
        registeredCards[cardCount].active = true;
        cardCount++;
        
        // Guardar en EEPROM
        saveCardsToEEPROM();
        
        Serial.println("Tarjeta registrada exitosamente!");
        Serial.print("UID: ");
        printUID(cardUID);
        Serial.println();
        Serial.print("Nombre: ");
        Serial.println(cardName);
    } else {
        Serial.println("Error: Máximo número de tarjetas alcanzado (50)");
    }
    
    mfrc522.PICC_HaltA();
}

void listRegisteredCards() {
    Serial.println("\\n=== TARJETAS REGISTRADAS ===");
    
    if (cardCount == 0) {
        Serial.println("No hay tarjetas registradas");
        return;
    }
    
    for (int i = 0; i < cardCount; i++) {
        if (registeredCards[i].active) {
            Serial.print("ID: ");
            Serial.print(i);
            Serial.print(" | UID: ");
            printUID(registeredCards[i].uid);
            Serial.print(" | Nombre: ");
            Serial.println(registeredCards[i].name);
        }
    }
    Serial.println("============================\\n");
}

bool isCardRegistered(byte* uid) {
    for (int i = 0; i < cardCount; i++) {
        if (registeredCards[i].active && 
            memcmp(registeredCards[i].uid, uid, 4) == 0) {
            return true;
        }
    }
    return false;
}

void printUID(byte* uid) {
    for (byte i = 0; i < 4; i++) {
        if (uid[i] < 0x10) Serial.print("0");
        Serial.print(uid[i], HEX);
        if (i < 3) Serial.print(":");
    }
}

void saveCardsToEEPROM() {
    EEPROM.put(0, cardCount);
    
    for (int i = 0; i < cardCount; i++) {
        int address = 4 + (i * sizeof(CardData));
        EEPROM.put(address, registeredCards[i]);
    }
    
    EEPROM.commit();
}

void loadCardsFromEEPROM() {
    EEPROM.get(0, cardCount);
    
    // Validar el contador
    if (cardCount < 0 || cardCount > 50) {
        cardCount = 0;
        return;
    }
    
    for (int i = 0; i < cardCount; i++) {
        int address = 4 + (i * sizeof(CardData));
        EEPROM.get(address, registeredCards[i]);
    }
    
    Serial.print("Cargadas ");
    Serial.print(cardCount);
    Serial.println(" tarjetas desde EEPROM");
}

/*

// Configuración de pines
#define RST_PIN    22
#define SS_PIN     5
#define LED_GREEN  2
#define LED_RED    4
#define BUZZER     15

// Configuración WiFi
const char* ssid = "TU_RED_WIFI";
const char* password = "TU_PASSWORD";

// Configuración del servidor web
WebServer server(80);

// Inicialización RFID
MFRC522 mfrc522(SS_PIN, RST_PIN);

// Variables del sistema
bool systemActive = true;
unsigned long lastCardTime = 0;
const unsigned long CARD_TIMEOUT = 2000; // 2 segundos entre lecturas

// Estructura para logs de acceso
struct AccessLog {
    byte uid[4];
    unsigned long timestamp;
    bool granted;
};

// Array para almacenar logs (últimos 100 accesos)
AccessLog accessLogs[100];
int logCount = 0;

void setup() {
    Serial.begin(115200);
    
    // Configurar pines
    pinMode(LED_GREEN, OUTPUT);
    pinMode(LED_RED, OUTPUT);
    pinMode(BUZZER, OUTPUT);
    
    // LEDs y buzzer apagados inicialmente
    digitalWrite(LED_GREEN, LOW);
    digitalWrite(LED_RED, LOW);
    digitalWrite(BUZZER, LOW);
    
    // Inicializar SPI y RFID
    SPI.begin();
    mfrc522.PCD_Init();
    
    // Inicializar EEPROM
    EEPROM.begin(4096);
    
    // Conectar WiFi
    connectToWiFi();
    
    // Configurar rutas del servidor web
    setupWebServer();
    
    // Iniciar servidor
    server.begin();
    
    Serial.println("Sistema RFID ESP32 iniciado!");
    Serial.print("Dirección IP: ");
    Serial.println(WiFi.localIP());
    
    // Señal de inicio (3 beeps cortos)
    for (int i = 0; i < 3; i++) {
        digitalWrite(BUZZER, HIGH);
        delay(100);
        digitalWrite(BUZZER, LOW);
        delay(100);
    }
}

void loop() {
    // Manejar solicitudes del servidor web
    server.handleClient();
    
    // Verificar si el sistema está activo
    if (!systemActive) {
        delay(100);
        return;
    }
    
    // Verificar tarjetas RFID
    if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
        
        // Evitar lecturas múltiples
        if (millis() - lastCardTime < CARD_TIMEOUT) {
            mfrc522.PICC_HaltA();
            return;
        }
        
        lastCardTime = millis();
        
        // Leer UID de la tarjeta
        byte cardUID[4];
        for (byte i = 0; i < 4; i++) {
            cardUID[i] = mfrc522.uid.uidByte[i];
        }
        
        // Procesar tarjeta
        bool accessGranted = processCard(cardUID);
        
        // Registrar en log
        logAccess(cardUID, accessGranted);
        
        // Respuesta visual y sonora
        if (accessGranted) {
            grantAccess();
        } else {
            denyAccess();
        }
        
        mfrc522.PICC_HaltA();
        
        // Imprimir información en serial
        Serial.print("Tarjeta: ");
        printUID(cardUID);
        Serial.print(" - Acceso: ");
        Serial.println(accessGranted ? "CONCEDIDO" : "DENEGADO");
    }
    
    delay(50);
}

void connectToWiFi() {
    Serial.print("Conectando a WiFi");
    WiFi.begin(ssid, password);
    
    int attempts = 0;
    while (WiFi.status() != WL_CONNECTED && attempts < 20) {
        delay(500);
        Serial.print(".");
        attempts++;
    }
    
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println();
        Serial.println("WiFi conectado!");
        Serial.print("IP: ");
        Serial.println(WiFi.localIP());
    } else {
        Serial.println();
        Serial.println("Error: No se pudo conectar a WiFi");
    }
}

void grantAccess() {
    // LED verde encendido
    digitalWrite(LED_GREEN, HIGH);
    digitalWrite(LED_RED, LOW);
    
    // Dos beeps cortos
    for (int i = 0; i < 2; i++) {
        digitalWrite(BUZZER, HIGH);
        delay(200);
        digitalWrite(BUZZER, LOW);
        delay(100);
    }
