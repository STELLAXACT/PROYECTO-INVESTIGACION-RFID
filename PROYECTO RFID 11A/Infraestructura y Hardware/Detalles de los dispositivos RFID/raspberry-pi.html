#!/usr/bin/env python3
"""
Sistema de Asistencia RFID para Raspberry Pi
Utiliza el m√≥dulo RC522 y pantalla LCD 16x2
"""

import RPi.GPIO as GPIO
import mysql.connector
from mfrc522 import SimpleMFRC522
import time
from datetime import datetime
import sys

# Configuraci√≥n de pines LCD
LCD_RS = 4
LCD_EN = 17
LCD_D4 = 27
LCD_D5 = 22
LCD_D6 = 23
LCD_D7 = 24

class AttendanceSystem:
    def __init__(self):
        """Inicializa el sistema de asistencia"""
        self.reader = SimpleMFRC522()
        self.setup_lcd()
        self.setup_database()
        
    def setup_lcd(self):
        """Configura la pantalla LCD"""
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(LCD_EN, GPIO.OUT)
        GPIO.setup(LCD_RS, GPIO.OUT)
        GPIO.setup(LCD_D4, GPIO.OUT)
        GPIO.setup(LCD_D5, GPIO.OUT)
        GPIO.setup(LCD_D6, GPIO.OUT)
        GPIO.setup(LCD_D7, GPIO.OUT)
        
        # Inicializar LCD
        self.lcd_init()
        
    def setup_database(self):
        """Configura la conexi√≥n a la base de datos"""
        try:
            self.db = mysql.connector.connect(
                host="localhost",
                user="root",
                password="tu_password",  # Cambiar por tu password
                database="attendance_system"
            )
            self.cursor = self.db.cursor()
            print("Conexi√≥n a base de datos exitosa")
        except mysql.connector.Error as err:
            print(f"Error de base de datos: {err}")
            sys.exit(1)
            
    def lcd_init(self):
        """Inicializa la pantalla LCD"""
        # Secuencia de inicializaci√≥n LCD
        GPIO.output(LCD_EN, False)
        GPIO.output(LCD_RS, False)
        time.sleep(0.01)
        
        # Comandos de inicializaci√≥n
        self.lcd_byte(0x33, False)
        self.lcd_byte(0x32, False)
        self.lcd_byte(0x28, False)
        self.lcd_byte(0x0C, False)
        self.lcd_byte(0x06, False)
        self.lcd_byte(0x01, False)
        
    def lcd_byte(self, bits, mode):
        """Env√≠a un byte a la pantalla LCD"""
        GPIO.output(LCD_RS, mode)
        
        # Bits superiores
        GPIO.output(LCD_D4, False)
        GPIO.output(LCD_D5, False)
        GPIO.output(LCD_D6, False)
        GPIO.output(LCD_D7, False)
        
        if bits & 0x10 == 0x10:
            GPIO.output(LCD_D4, True)
        if bits & 0x20 == 0x20:
            GPIO.output(LCD_D5, True)
        if bits & 0x40 == 0x40:
            GPIO.output(LCD_D6, True)
        if bits & 0x80 == 0x80:
            GPIO.output(LCD_D7, True)
            
        self.lcd_toggle_enable()
        
        # Bits inferiores
        GPIO.output(LCD_D4, False)
        GPIO.output(LCD_D5, False)
        GPIO.output(LCD_D6, False)
        GPIO.output(LCD_D7, False)
        
        if bits & 0x01 == 0x01:
            GPIO.output(LCD_D4, True)
        if bits & 0x02 == 0x02:
            GPIO.output(LCD_D5, True)
        if bits & 0x04 == 0x04:
            GPIO.output(LCD_D6, True)
        if bits & 0x08 == 0x08:
            GPIO.output(LCD_D7, True)
            
        self.lcd_toggle_enable()
        
    def lcd_toggle_enable(self):
        """Activa el pin enable del LCD"""
        time.sleep(0.0005)
        GPIO.output(LCD_EN, True)
        time.sleep(0.0005)
        GPIO.output(LCD_EN, False)
        time.sleep(0.0005)
        
    def lcd_string(self, message, line):
        """Muestra un string en la pantalla LCD"""
        message = message.ljust(16, " ")
        
        self.lcd_byte(line, False)
        
        for i in range(16):
            self.lcd_byte(ord(message[i]), True)
            
    def get_user_by_rfid(self, rfid_uid):
        """Busca un usuario por su UID RFID"""
        query = "SELECT id, name FROM users WHERE rfid_uid = %s"
        self.cursor.execute(query, (str(rfid_uid),))
        return self.cursor.fetchone()
        
    def record_attendance(self, user_id):
        """Registra la asistencia del usuario"""
        # Verificar si ya hay un check-in sin check-out
        query = "SELECT id FROM attendance WHERE user_id = %s AND check_out IS NULL"
        self.cursor.execute(query, (user_id,))
        active_session = self.cursor.fetchone()
        
        if active_session:
            # Hacer check-out
            query = "UPDATE attendance SET check_out = %s WHERE id = %s"
            self.cursor.execute(query, (datetime.now(), active_session[0]))
            return "check_out"
        else:
            # Hacer check-in
            query = "INSERT INTO attendance (user_id) VALUES (%s)"
            self.cursor.execute(query, (user_id,))
            return "check_in"
            
        self.db.commit()
        
    def run(self):
        """Ejecuta el bucle principal del sistema"""
        print("Sistema de Asistencia RFID iniciado")
        self.lcd_string("Sistema RFID", 0x80)
        self.lcd_string("Listo...", 0xC0)
        
        try:
            while True:
                print("Acerca una tarjeta RFID...")
                
                # Leer tarjeta RFID
                rfid_uid, text = self.reader.read()
                
                print(f"Tarjeta detectada: {rfid_uid}")
                
                # Buscar usuario
                user = self.get_user_by_rfid(rfid_uid)
                
                if user:
                    user_id, name = user
                    action = self.record_attendance(user_id)
                    
                    # Mostrar en LCD
                    self.lcd_string(name[:16], 0x80)
                    
                    if action == "check_in":
                        self.lcd_string("Entrada: " + datetime.now().strftime("%H:%M"), 0xC0)
                        print(f"Check-in: {name}")
                    else:
                        self.lcd_string("Salida: " + datetime.now().strftime("%H:%M"), 0xC0)
                        print(f"Check-out: {name}")
                        
                else:
                    # Usuario no encontrado
                    self.lcd_string("Usuario no", 0x80)
                    self.lcd_string("encontrado", 0xC0)
                    print("Usuario no registrado")
                    
                time.sleep(3)
                
                # Limpiar pantalla
                self.lcd_string("Sistema RFID", 0x80)
                self.lcd_string("Listo...", 0xC0)
                
        except KeyboardInterrupt:
            print("\nSistema detenido por el usuario")
        finally:
            self.cleanup()
            
    def cleanup(self):
        """Limpia los recursos utilizados"""
        GPIO.cleanup()
        if hasattr(self, 'db'):
            self.db.close()
        print("Recursos liberados")

if __name__ == "__main__":
    system = AttendanceSystem()
    system.run()
